## ✅ Complete Python CLI Template

```python id="allinone_cli"
#!/usr/bin/env python3

import argparse

def main():
    parser = argparse.ArgumentParser(
        description="Comprehensive CLI template with flags, options, and positional arguments."
    )

    # ---------------------------
    # Positional arguments
    # ---------------------------
    parser.add_argument(
        "input1",
        type=str,
        help="First input (single value)"
    )

    parser.add_argument(
        "input2",
        type=str,
        nargs="+",
        help="Second input(s), one or more values"
    )

    # ---------------------------
    # Optional flags and options
    # ---------------------------

    # Boolean flag
    parser.add_argument(
        "-v", "--verbose",
        action="store_true",
        help="Enable verbose mode"
    )

    # Optional single-value option
    parser.add_argument(
        "-o", "--output",
        type=str,
        default="output.txt",
        help="Output file (default: output.txt)"
    )

    # Optional option with multiple integer values
    parser.add_argument(
        "-n", "--numbers",
        type=int,
        nargs="+",
        help="One or more integer numbers"
    )

    # Optional option with multiple string values
    parser.add_argument(
        "-s", "--strings",
        type=str,
        nargs="+",
        help="One or more string values"
    )

    args = parser.parse_args()

    # ---------------------------
    # Example usage
    # ---------------------------
    print("=== Parsed Arguments ===")
    print(f"Input 1: {args.input1}")
    print(f"Input 2: {args.input2}")
    print(f"Output: {args.output}")
    print(f"Verbose: {args.verbose}")
    print(f"Numbers: {args.numbers}")
    print(f"Strings: {args.strings}")

if __name__ == "__main__":
    main()
```
